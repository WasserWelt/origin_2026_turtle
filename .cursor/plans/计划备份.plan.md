# 底盘舵轮改全向轮 + 云台大Pitch轴 修改计划

## 一、修改概览

### 电机通信布局变更

| 通信线路 | 原有配置 | 新配置 |
|---------|---------|--------|
| **CAN1** (上下板通信) | 大Yaw(DM6006)、小Yaw(GM6020)、4个舵电机(GM6020)、4个轮电机(3508) | 大Yaw(DM6006)、小Yaw(GM6020)、**大Pitch(DM4340)** |
| **CAN2-云台上板** | 摩擦轮x2(3508)、拨弹盘(MF4010)、小Pitch(GM6020) | 摩擦轮x2(3508)、拨弹盘(MF4010)、**小Pitch(MF6015)** |
| **CAN2-底盘下板** | 无 | **轮电机x4(3508)** |

### 关键变更
- **底盘**: 8个电机(4舵+4轮) → 4个电机(全向轮3508)
- **云台**: 增加大Pitch轴(DM4340)，用于抬起/降下整个云台
- **通信**: 底盘下板新增独立CAN2通道用于4个轮电机

### 2.9 全向轮运动学解算 (`Chassis_Task.c`)

**参考Mecanum项目的运动学公式，但保持原始项目代码风格**：

```c
/**
 * @description: 通过底盘坐标系下的目标速度解算出四个麦克纳姆轮的目标速度
 * @param vx_set 底盘坐标系下x轴目标速度 (rpm)
 * @param vy_set 底盘坐标系下y轴目标速度 (rpm)
 * @param wz_set 底盘坐标系下z轴目标角速度 (rpm)
 * @note 麦克纳姆轮排布:
 *   wheel_speed[0] = 右前轮 (FR)
 *   wheel_speed[1] = 左前轮 (FL)
 *   wheel_speed[2] = 右后轮 (BR)
 *   wheel_speed[3] = 左后轮 (BL)
 */
static void Chassis_Vector_To_Mecanum_Speed(const fp32 vx_set, const fp32 vy_set, const fp32 wz_set)
{
    fp32 wheel_speed[4];

    wheel_speed[WHEEL_MOTOR_FR] = (vx_set + vy_set) + (CHASSIS_WZ_SET_SCALE + 1.0f) * MOTOR_DISTANCE_TO_CENTER * wz_set;
    wheel_speed[WHEEL_MOTOR_FL] = (vx_set - vy_set) + (-CHASSIS_WZ_SET_SCALE + 1.0f) * MOTOR_DISTANCE_TO_CENTER * wz_set;
    wheel_speed[WHEEL_MOTOR_BR] = (-vx_set + vy_set) + (-CHASSIS_WZ_SET_SCALE + 1.0f) * MOTOR_DISTANCE_TO_CENTER * wz_set;
    wheel_speed[WHEEL_MOTOR_BL] = (-vx_set - vy_set) + (CHASSIS_WZ_SET_SCALE + 1.0f) * MOTOR_DISTANCE_TO_CENTER * wz_set;

    for (uint8_t i = 0; i < 4; i++)
    {
        chassis_m3508[i].speed_set = wheel_speed[i];
    }
}
```

### 2.13 功率控制 (`Chassis_Power_Limitor.c`)

**参考Mecanum项目简化实现**：

```c
// 电机功率模型系数
fp32 toque_coefficient = 1.99688994e-6f;
fp32 k1 = 1.23e-07;
fp32 k2 = 1.453e-07;
fp32 constant = 4.081f;

void Chassis_Power_Control(power_limitor_t *limiter, chassis_motor_t *motors, fp32 chassis_max_power)
{
    fp32 initial_total_power = 0;
    fp32 initial_give_power[4];

    // 计算各电机实际消耗功率
    for (uint8_t i = 0; i < 4; i++)
    {
        initial_give_power[i] = toque_coefficient * motors[i].give_current * motors[i].speed +
                               k1 * motors[i].give_current * motors[i].give_current +
                               k2 * motors[i].speed * motors[i].speed + constant;

        if (initial_give_power[i] < 0)
            continue;
        initial_total_power += initial_give_power[i];
    }

    // 功率超限时按比例缩放
    if (initial_total_power > chassis_max_power)
    {
        fp32 power_scale = chassis_max_power / initial_total_power;
        for (uint8_t i = 0; i < 4; i++)
        {
            if (initial_give_power[i] < 0)
                continue;

            fp32 b = toque_coefficient * motors[i].speed;
            fp32 c = k2 * motors[i].speed * motors[i].speed - initial_give_power[i] * power_scale + constant;

            if (motors[i].give_current > 0)
            {
                fp32 temp = (-b + sqrt(b * b - 4 * k1 * c)) / (2 * k1);
                motors[i].give_current = (temp > 16000) ? 16000 : (int16_t)temp;
            }
            else
            {
                fp32 temp = (-b - sqrt(b * b - 4 * k1 * c)) / (2 * k1);
                motors[i].give_current = (temp < -16000) ? -16000 : (int16_t)temp;
            }
        }
    }
}
```

### 3.6 大Pitch控制逻辑 (`Gimbal_Task.c`)

**使用Q键切换抬起/落下状态，保持原始项目风格**：

```c
/*******************************大Pitch控制相关********************************/
typedef enum
{
    BIG_PITCH_UP,        // 抬起状态
    BIG_PITCH_DOWN,      // 落下状态
} big_pitch_state_t;

static big_pitch_state_t big_pitch_state = BIG_PITCH_DOWN;  // 默认落下
static fp32 big_pitch_target_angle = 0.0f;

// 大Pitch控制相关函数声明
static void Big_Pitch_Control_Init(void);
static void Big_Pitch_Data_Update(void);
static void Big_Pitch_Mode_Handler(void);
static void Big_Pitch_Motor_Control(void);
static void Gimbal_Motor_Control_Init(void);  // 原有函数声明
static void Gimbal_Data_Update(void);          // 原有函数声明
static gimbal_mode_t Gimbal_Mode_Update(void); // 原有函数声明
static void Call_Gimbal_Mode_Handler(gimbal_mode_t mode);  // 原有函数声明

/*******************************大Pitch初始化********************************/
static void Big_Pitch_Control_Init(void)
{
    const static fp32 big_pitch_speed_pid[3] = {BIG_PITCH_MOTOR_SPEED_PID_KP, BIG_PITCH_MOTOR_SPEED_PID_KI, BIG_PITCH_MOTOR_SPEED_PID_KD};
    const static fp32 big_pitch_angle_pid[3] = {BIG_PITCH_MOTOR_ANGLE_PID_KP, BIG_PITCH_MOTOR_ANGLE_PID_KI, BIG_PITCH_MOTOR_ANGLE_PID_KD};

    PID_init(&DM_big_pitch_motor.speed_pid, PID_POSITION, big_pitch_speed_pid,
             BIG_PITCH_MOTOR_SPEED_PID_MAX_OUT, BIG_PITCH_MOTOR_SPEED_PID_MAX_IOUT);
    PID_init(&DM_big_pitch_motor.angle_pid, PID_POSITION, big_pitch_angle_pid,
             BIG_PITCH_MOTOR_ANGLE_PID_MAX_OUT, BIG_PITCH_MOTOR_ANGLE_PID_MAX_IOUT);
}

/*******************************大Pitch数据更新********************************/
static void Big_Pitch_Data_Update(void)
{
    // MIT协议数据解析已在CAN接收回调中完成
    // angle_now 和 speed_now 在CAN回调中更新
    DM_big_pitch_motor.angle_now = DM_big_pitch_motor.pos * 57.3248f; // 转度
    DM_big_pitch_motor.speed_now = DM_big_pitch_motor.vel;
}

/*******************************大Pitch模式处理********************************/
static void Big_Pitch_Mode_Handler(void)
{
    // 1. Q键切换抬起/落下状态 (按键按下触发)
    static uint8_t last_key_q = 0;
    if ((rc_ctrl.key.v & KEY_PRESSED_OFFSET_Q) && !last_key_q)
    {
        if (big_pitch_state == BIG_PITCH_DOWN)
            big_pitch_state = BIG_PITCH_UP;
        else
            big_pitch_state = BIG_PITCH_DOWN;
    }
    last_key_q = (rc_ctrl.key.v & KEY_PRESSED_OFFSET_Q) ? 1 : 0;

    // 2. ch[4]打杆触发切换 (拨杆从中间打到一端再回中触发一次)
    static int8_t last_ch4_state = 0;  // 0: 中间, 1: 上/下端
    int8_t current_ch4_state = (rc_ctrl.rc.ch[4] > 300) ? 1 : ((rc_ctrl.rc.ch[4] < -300) ? -1 : 0);

    if (last_ch4_state != 0 && current_ch4_state == 0)
    {
        // 从非中间状态回到中间，触发一次切换
        if (big_pitch_state == BIG_PITCH_DOWN)
            big_pitch_state = BIG_PITCH_UP;
        else
            big_pitch_state = BIG_PITCH_DOWN;
    }
    last_ch4_state = current_ch4_state;

    // 3. 根据状态设置目标角度
    switch (big_pitch_state)
    {
    case BIG_PITCH_UP:
        big_pitch_target_angle = BIG_PITCH_ANGLE_MAX;
        break;
    case BIG_PITCH_DOWN:
        big_pitch_target_angle = BIG_PITCH_ANGLE_MIN;
        break;
    }

    DM_big_pitch_motor.angle_now = big_pitch_target_angle;
}

/*******************************大Pitch电机控制********************************/
static void Big_Pitch_Motor_Control(void)
{
    if (gimbal_control.gimbal_mode == GIMBAL_SAFE)
    {
        DM_big_pitch_motor.target_torque = 0;
        return;
    }

    // 角度环
    PID_calc(&DM_big_pitch_motor.angle_pid, DM_big_pitch_motor.angle_now, big_pitch_target_angle);

    // 速度环
    DM_big_pitch_motor.target_torque = DM_big_pitch_motor.angle_pid.out;
    DM_big_pitch_motor.target_torque = limit(DM_big_pitch_motor.target_torque, -10.0f, 10.0f);
}
```

### 3.10 大Pitch CAN发送

由于CAN1总线上有多种格式的报文（标准CAN + MIT协议），**大Pitch的CAN发送应在云台主任务中直接发送，或在Can_Timer中根据报文类型分别处理**。

**方案A**: 在云台主任务中直接发送（推荐）

```c
// 在Gimbal_Task.c中添加
static void Big_Pitch_CAN_Send(void)
{
    CanTxMsgTypeDef tx_msg;
    tx_msg.StdId = BIG_PITCH_DM4340_CMD;
    tx_msg.IDE = CAN_ID_STD;
    tx_msg.RTR = CAN_RTR_DATA;
    tx_msg.DLC = 0x08;

    int16_t pos = float_to_uint(DM_big_pitch_motor.angle_now / 57.3248f, P_MIN, P_MAX, 16);
    int16_t vel = float_to_uint(0, V_MIN, V_MAX, 12);  // 速度设为0，使用角度环
    int16_t kp = float_to_uint(10.0f, KP_MIN, KP_MAX, 12);
    int16_t kd = float_to_uint(1.0f, KD_MIN, KD_MAX, 12);
    int16_t torq = float_to_uint(DM_big_pitch_motor.target_torque, T_MIN, T_MAX, 12);

    tx_msg.Data[0] = (pos >> 8) & 0xFF;
    tx_msg.Data[1] = pos & 0xFF;
    tx_msg.Data[2] = (vel >> 4) & 0xFF;
    tx_msg.Data[3] = ((vel & 0xF) << 4) | ((kp >> 8) & 0xF);
    tx_msg.Data[4] = kp & 0xFF;
    tx_msg.Data[5] = (kd >> 4) & 0xFF;
    tx_msg.Data[6] = ((kd & 0xF) << 4) | ((torq >> 8) & 0xF);
    tx_msg.Data[7] = torq & 0xFF;

    // 直接发送或放入队列
    xQueueSend(CAN1_send_queue, &tx_msg, 0);
}
```
