# 底盘舵轮改全向轮 + 云台大Pitch轴 修改计划

## 一、修改概览

### 电机通信布局变更


| 通信线路             | 原有配置                                                | 新配置                                          |
| ---------------- | --------------------------------------------------- | -------------------------------------------- |
| **CAN1** (上下板通信) | 大Yaw(DM6006)、小Yaw(GM6020)、4个舵电机(GM6020)、4个轮电机(3508) | 大Yaw(DM6006)、小Yaw(GM6020)、**大Pitch(DM4340)** |
| **CAN2-云台上板**    | 摩擦轮x2(3508)、拨弹盘(MF4010)、小Pitch(GM6020)              | 摩擦轮x2(3508)、拨弹盘(MF4010)、**小Pitch(MF6015)**   |
| **CAN2-底盘下板**    | 无                                                   | **轮电机x4(3508)**                              |


### 关键变更

- **底盘**: 8个电机(4舵+4轮) → 4个电机(全向轮3508)
- **云台**: 增加大Pitch轴(DM4340)，用于抬起/降下整个云台
- **通信**: 底盘下板新增独立CAN2通道用于4个轮电机

#### E. 全向轮运动学解算 (`Chassis_Task.c`)

```c
// ============== 核心修改: 舵轮控制逻辑 → 全向轮控制逻辑 ==============

// 麦克纳姆轮运动学 (3种常见安装方式)
// 方式1: X型安装 (推荐)
//   \\ /     / \\        前: 电机1, 后: 电机2
//    \\     /   \\        左: 电机3, 右: 电机4
//   /  \\ /     \\  /
  
// 麦克纳姆轮速度解算公式
// v1 = vx - vy - wz*(L+W)  // 左前轮
// v2 = vx + vy + wz*(L+W)  // 右前轮  
// v3 = vx + vy - wz*(L+W)  // 左后轮
// v4 = vx - vy + wz*(L+W)  // 右后轮

static void Omniwheel_Vector_To_Speed(fp32 vx_set, fp32 vy_set, fp32 wz_set) {
    // 底盘尺寸参数 (需要重新测量)
    #define WHEEL_BASE_LENGTH 0.30f   // 前后轮距离 (m)
    #define WHEEL_BASE_WIDTH  0.25f   // 左右轮距离 (m)
    
    fp32 r = sqrtf(WHEEL_BASE_LENGTH*WHEEL_BASE_LENGTH + WHEEL_BASE_WIDTH*WHEEL_BASE_WIDTH);
    
    // 速度解算 (rpm)
    wheel_speed_target[0] =  vx_set - vy_set - wz_set * r;  // 左前
    wheel_speed_target[1] =  vx_set + vy_set + wz_set * r;  // 右前
    wheel_speed_target[2] =  vx_set + vy_set - wz_set * r;  // 左后
    wheel_speed_target[3] =  vx_set - vy_set + wz_set * r;  // 右后
    
    // 转换为电机转速 (rpm)
    for (uint8_t i = 0; i < 4; i++) {
        // 线速度(m/s) 转 电机转速(rpm)
        // v_rpm = v_mps / (2*π*R) * 60 / reduction_ratio
        wheel_speed_target[i] = wheel_speed_target[i] * 60.0f / (2.0f * PI * WHEEL_RADIUS);
        wheel_speed_target[i] = wheel_speed_target[i] / REDUCTION_RATIO;
        
        omni_wheel_motor[i].speed_set = wheel_speed_target[i];
    }
}

// ============== 删除: 舵轮相关函数 ==============

// static fp32 Find_Steer_Min_Angle(...) → 删除
```

```c
// ============== 修改: 串级PID → 单级速度环 ==============
static void Chassis_Motor_Control_Current_Set(chassis_mode_t mode) {
    if (mode == CHASSIS_SAFE) return;
    
    for (uint8_t i = 0; i < 4; i++) {
        // 全向轮只需速度环PID
        PID_calc(&omni_wheel_motor[i].speed_pid,
                 omni_wheel_motor[i].speed_now,
                 omni_wheel_motor[i].speed_set);
        
        // 前馈控制
        fp32 ff = WHEEL_SPEED_FF * (omni_wheel_motor[i].speed_set - 
                                     omni_wheel_motor[i].speed_set_last);
        
        omni_wheel_motor[i].give_current = (int16_t)(omni_wheel_motor[i].speed_pid.out + ff);
    }
}
```

```c
static void Chassis_Mode_Handler(chassis_mode_t mode) {

        case FOLLOW_GIMBAL:
            // 跟随云台模式 - 计算底盘相对于云台的目标位置
```

```c
static void Power_Calc_Motor_Model(power_limitor_t *limiter, ...) {
    // 仅计算4个轮电机的功率消耗
    for (int i = 0; i < 4; i++) {
```

```c
// ============== 新增: 大Pitch控制模式枚举 ==============
typedef enum {
    BIG_PITCH_UP,        // 抬起云台
    BIG_PITCH_DOWN,      // 降下云台
    BIG_PITCH_HOLD,      // 保持当前位置
    BIG_PITCH_REMOTE,    // 遥控器手动控制
    BIG_PITCH_SAFE       // 失能
} big_pitch_mode_t;

// ============== 新增: 大Pitch模式更新 ==============
static big_pitch_mode_t Big_Pitch_Mode_Update(void) {
    // 拨杆控制大Pitch升降
    if (rc_ctrl.rc.s[1] == RC_SW_DOWN) {
        return BIG_PITCH_SAFE;  // 紧急失能
    }
    
    if (rc_ctrl.key.v & KEY_PRESSED_OFFSET_Q) {
        return BIG_PITCH_UP;    // Q键抬起
    }
    
    if (rc_ctrl.key.v & KEY_PRESSED_OFFSET_E) {
        return BIG_PITCH_DOWN;  // E键降下
    }
    
    if (abs(rc_ctrl.rc.ch[4]) > 100) {
        return BIG_PITCH_REMOTE; // 通道控制
    }
    
    return BIG_PITCH_HOLD;
}

// ============== 新增: 大Pitch模式处理 ==============
static void Big_Pitch_Mode_Handler(big_pitch_mode_t mode) {
    static fp32 target_angle = 0.0f;
    
    switch(mode) {
        case BIG_PITCH_UP:
            target_angle = BIG_PITCH_ANGLE_MAX;
            break;
            
        case BIG_PITCH_DOWN:
            target_angle = BIG_PITCH_ANGLE_MIN;
            break;
            
        case BIG_PITCH_REMOTE:
            // 通道4控制大Pitch速度
            target_angle += rc_ctrl.rc.ch[4] / 660.0f * BIG_PITCH_RC_CONTROL_SPEED * 0.002f;
            // 限幅
            target_angle = LIMIT(target_angle, BIG_PITCH_ANGLE_MIN, BIG_PITCH_ANGLE_MAX);
            break;
            
        case BIG_PITCH_HOLD:
            target_angle = DM_big_pitch_motor.angle_now;
            break;
            
        case BIG_PITCH_SAFE:
            target_angle = DM_big_pitch_motor.angle_now;
            break;
    }
    
    DM_big_pitch_motor.INS_angle_set = target_angle;
}

// ============== 新增: 大Pitch PID计算 ==============
static void Big_Pitch_Motor_Control(void) {
    if (gimbal_control.gimbal_mode == GIMBAL_SAFE) {
        DM_big_pitch_motor.target_torque = 0;
        return;
    }
    
    // 角度环
    PID_calc(&DM_big_pitch_motor.angle_pid,
             DM_big_pitch_motor.angle_now,
             DM_big_pitch_motor.INS_angle_set);
    
    // 速度环
    DM_big_pitch_motor.INS_speed_set = DM_big_pitch_motor.angle_pid.out;
    PID_calc(&DM_big_pitch_motor.speed_pid,
             DM_big_pitch_motor.speed_now,
             DM_big_pitch_motor.INS_speed_set);
    
    DM_big_pitch_motor.target_torque = DM_big_pitch_motor.speed_pid.out;
}
```

```c
void CAN1_Timer_Callback(TimerHandle_t xTimer) {
    CanTxMsgTypeDef tx_msg;
    
    // 发送大Pitch DM4340控制
    // MIT协议: [ID][pos_h][pos_l][vel_h][vel_l][kp_h][kp_l][kd_h][kd_l][torq_h][torq_l]
    // 注意: DM4340可能使用不同的MIT格式
    
    tx_msg.StdId = BIG_PITCH_DM4340_CMD;
    tx_msg.DLC = 8;
    
    // 角度转换
    int16_t pos = float_to_uint(DM_big_pitch_motor.INS_angle_set / 57.3248f, P_MIN, P_MAX, 16);
    int16_t vel = float_to_uint(DM_big_pitch_motor.INS_speed_set, V_MIN, V_MAX, 12);
    int16_t kp = float_to_uint(10.0f, KP_MIN, KP_MAX, 12);  // 固定KP
    int16_t kd = float_to_uint(1.0f, KD_MIN, KD_MAX, 12);    // 固定KD
    int16_t torq = float_to_uint(DM_big_pitch_motor.target_torque, T_MIN, T_MAX, 12);
    
    tx_msg.Data[0] = (pos >> 8) & 0xFF;
    tx_msg.Data[1] = pos & 0xFF;
    tx_msg.Data[2] = (vel >> 4) & 0xFF;
    tx_msg.Data[3] = ((vel & 0xF) << 4) | ((kp >> 8) & 0xF);
    tx_msg.Data[4] = kp & 0xFF;
    tx_msg.Data[5] = (kd >> 4) & 0xFF;
    tx_msg.Data[6] = ((kd & 0xF) << 4) | ((torq >> 8) & 0xF);
    tx_msg.Data[7] = torq & 0xFF;
    
    xQueueSend(CAN1_send_queue, &tx_msg, 0);
}
```
